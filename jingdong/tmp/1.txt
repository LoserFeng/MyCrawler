	    time = time || 3600
	    path = path || '/'
	    var str = name + '=' + value + '; '
	    if(time) str += 'expires=' + new Date(Date.now() + time * 1000).toGMTString() + '; '
	    if(path) str += 'path=' + path + '; '
	    if(domain == null){
            var host = window.location.hostname.split('.')
            host[0] = ''
            domain = host.join('.')
	    }
	    str += 'domain=' + domain + "; SameSite=None; Secure=true"
	    document.cookie = str;
}
